49void initPWMasPIT(uint8_t slice, uint16_t milis, bool enable)
50{
51 assert(milis<=262); // PWM can manage interrupt periods greater than 262 milis
52 float prescaler = (float)SYS_CLK_KHZ/500;
53 assert(prescaler<256); // the integer part of the clock divider can be greater than 255
54 // || counter frecuency ||| Period in seconds taking into account de phase correct mode |||
55 uint32_t wrap = (1000*SYS_CLK_KHZ*milis/(prescaler*2*1000)); // 500000*milis/2000
56 assert(wrap<((1UL<<17)-1));
57 // Configuring the PWM
58 pwm_config cfg = pwm_get_default_config();
59 pwm_config_set_phase_correct(&cfg, true);
60 pwm_config_set_clkdiv(&cfg, prescaler);
61 pwm_config_set_clkdiv_mode(&cfg, PWM_DIV_FREE_RUNNING);
62 pwm_config_set_wrap(&cfg, wrap);
63 pwm_set_irq_enabled(slice, true);
64 irq_set_enabled(PWM_IRQ_WRAP, true);
65 pwm_init(slice, &cfg, enable);
66 }

68 void pwmIRQ(void)
69 {
70 uint32_t cols;
71 bool button;
72 switch (pwm_get_irq_status_mask())
73 {
74 case 0x01UL: // PWM slice 0 ISR used as a PIT to generate row sequence
75 kp_gen_seq(&gKeyPad);
76 pwm_clear_irq(0); // Acknowledge slice 0 PWM IRQ
77 break;
78
79 case 0x02UL: // PWM slice 1 ISR used as a PIT to implement the keypad debouncer
80 cols = gpio_get_all() & 0x000003C0; // Get columns gpio values
81 if(kp_is_2nd_zero(&gKeyPad)){
82 if(!cols){
83 kp_set_irq_enabled(&gKeyPad, true); // Enable the GPIO IRQs
84 pwm_set_enabled(0, true); // Enable the row sequence
85 pwm_set_enabled(1, false); // Disable the keypad debouncer
86 gKeyPad.KEY.dbnc = 0;
87 }
88 else
89 kp_clr_zflag(&gKeyPad);
90 }
91 else{
92 if(!cols)
93 kp_set_zflag(&gKeyPad);
94 }
95
96 pwm_clear_irq(1); // Acknowledge slice 1 PWM IRQ
97 break;
98
99 case 0x04UL: // PWM slice 2 ISR used as a PIT to implement the button debouncer
100 button = gpio_get(gButton.KEY.gpio_num);
101 if(button_is_2nd_zero(&gButton)){
102 if(!button){
103 signal_set_state(&gSignal, (gSignal.STATE.ss + 1)%4);
104 button_set_irq_enabled(&gButton, true); // Enable the GPIO IRQs
105 pwm_set_enabled(2, false); // Disable the button debouncer
106 signal_calculate(&gSignal); // Recalculate the signal values
107 gButton.KEY.dbnc = 0;
108 }
109 else
110 button_clr_zflag(&gButton);
111 }
112 else{
113 if(!button)
114 button_set_zflag(&gButton);
115 }
116 pwm_clear_irq(2); // Acknowledge slice 2 PWM IRQ
117 break;
118
119 default:
120 printf("Happend what should not happens on PWM IRQ\n");
121 break;
122 }
123 }

136 void keypadCallback(uint num, uint32_t mask)
137 {
138 // Capture the key pressed
139 uint32_t cols = gpio_get_all() & 0x000003C0; // Get columns gpio values
140 kp_capture(&gKeyPad, cols);
141 // printf("Key: %02x\n", gKeyPad.KEY.dkey);
142
143 pwm_set_enabled(0, false); // Disable the row sequence
144 pwm_set_enabled(1, true); // Enable the keypad debouncer
145 kp_set_irq_enabled(&gKeyPad, false); // Disable the keypad IRQs
146
147 kp_set_zflag(&gKeyPad); // Set the flag that indicates that a zero was detected on keypad
148 gKeyPad.KEY.dbnc = 1;
149
150 // Auxiliar variables
151 static uint8_t in_param_state = 0x00; // 0: Nothing, 1: Entering amp (A), 2: Entering offset (B), 3: Entering freq (C)
152 static uint32_t param = 0; // This variable will store the value of any parameter that is being entered
153 static uint8_t key_cont = 0;
154
155
156 // Process the key pressed
157
158 // To accept a number, in_param_state must be different of 0
159 if(checkNumber(gKeyPad.KEY.dkey) && in_param_state){
160 param = (!key_cont)? gKeyPad.KEY.dkey : param*10 + gKeyPad.KEY.dkey;
161 key_cont++;
162 }
163 // To accept a letter different of 0x0D, in_param_state must be 0
164 else if(checkLetter(gKeyPad.KEY.dkey) && !in_param_state){
165 // cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 1);
166 led_on(gLed);
167 switch (gKeyPad.KEY.dkey)
168 {
169 case 0x0A:
170 in_param_state = 1;
171 break;
172 case 0x0B:
173 in_param_state = 2;
174 break;
175 case 0x0C:
176 in_param_state = 3;
177 break;
178 default:
179 printf("Invalid letter\n");
180 break;
181 }
182 }
183 // To accept a 0x0D, in_param_state must be different of 0
184 else if(gKeyPad.KEY.dkey == 0x0D && in_param_state){
185 // cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 0);
186 led_off(gLed);
187 switch (in_param_state)
188 {
189 case 1:
190 if(checkAmp(param)){
191 signal_set_amp(&gSignal,param);
192 }
193 break;
194 case 2:
195 if(checkOffset(param)){
196 signal_set_offset(&gSignal,param);
197 }
198 break;
199 case 3:
200 if(checkFreq(param)){
201 signal_set_freq(&gSignal,param);
202 }
203 break;
204 default:
205 printf("Invalid state\n");
206 break;
207 }
208 signal_calculate(&gSignal);
209 in_param_state = 0;
210 param = 0;
211 key_cont = 0;
212 }
213 gKeyPad.KEY.nkey = 0;
214
215 gpio_acknowledge_irq(num, mask); // gpio IRQ acknowledge
216 }
